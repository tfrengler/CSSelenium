enum TestState
{
    NOT_STARTED,
    IN_PROGRESS,
    OK,
    NOK
}

abstract TestCase
- public const ID
- public const string Description
- Exception LastException {get; private set;}
- OpenQA.Selenium.Screenshot LastVisualState {get; private set;}
- enum State
- Started {get; private set;}
- Finished {get; private set;}
- public TimeTaken()
- private OnFailure()
- private OnFailedAssertion()
- abstract bool Setup()
- abstract bool Run()
- abstract bool Teardown()
- Constructor(SeleniumWrapper selenium, object pageObject, ILogger logger, TestData testdata)

sealed TestSuite
- List<TestCase> TestCases
- BeforeEach()
- AfterEach()
- BeforeAll()
- AfterAll()
- GetFailedTestCases()
- GetSuccessfulTestCases()
- Constructor(SeleniumWrapper selenium ILogger logger, List<TestCases> testCases, Browser browser, Server server)

sealed TestBatch
- List<TestCase> TestCases
- List<TestSuite> TestSuites
- Browser
- Server
- StopAfterFailure (the amount of failures after which to stop execution)
- Started
- Finished
- TimeTaken()
- Start()
- Stop()
- BeforeEach()
- AfterEach()
- BeforeAll()
- AfterAll()
- Constructor(SeleniumWrapper selenium ILogger logger, List<TestCases> testCases, Browser browser, Server server)
- Constructor(SeleniumWrapper selenium ILogger logger, List<TestSuite> testSuites, Browser browser, Server server)

TestData
- Interface for test cases to get test data without needing knowledge of the rest of the system
