enum Browser
{
    CHROME, FIREFOX, EDGE
}

SeleniumWrapper:
- Chosing Browser
- Remote- or Local-mode
    - Remote mode accepts a URL
- Browser arguments
    - If DriverOptions are passed ignore
- Accept DriverOptions or create default ones
- Default options:
    - Page load timeout = 120 seconds
    - Firefox: new profile created
    - Remote = false: LocalFileDetector
    - No proxy
    - PageLoad normal (entire page is loaded)

SeleniumWrapper(Browser browser, URI remoteURL)
SeleniumWrapper(Browser browser, URI remoteURL, string[] browserArguments)
SeleniumWrapper(Browser browser, URI remoteURL, Proxy proxy)
SeleniumWrapper(Browser browser, URI remoteURL, Proxy proxy, string[] browserArguments)
SeleniumWrapper(Browser browser, URI remoteURL, DriverOptions options)

static WebdriverWrapper:
- Knowledge of browser webdrivers
- Starts webdriver process
- Shuts down webdriver process

static WebdriverWrapper.Location (DirectoryInfo of where the browser drivers live)
static WebdriverWrapper.Kill(Browser browser)
static WebdriverWrapper.Start(Browser browser)
